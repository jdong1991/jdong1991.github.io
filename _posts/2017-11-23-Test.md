def kdTree(pointList):
    if len(pointList)==0:
        return
    pointList=np.array(pointList)
    m=pointList.shape[0]
    SelectDimensionalNo=np.var(pointList,axis=0).argmax(axis=0)
    medianNo=m//2
    sortPoint=np.argsort(pointList[:,SelectDimensionalNo])
    return (tuple(pointList[sortPoint[medianNo]]),kdTree([pointList[sortPoint[k]] for k in range(medianNo)])
            ,kdTree([pointList[sortPoint[k]] for k in range(medianNo+1,m)]))
